Expressions

The Hulo AuthorsMarch 15, 2025Less than 1 minute

关键字

typeof

echo(typeof 10) // num

let i: int = 1
echo(typeof i == num) // true

is

let a = 10
if a is num {
    echo "a is a number"
}

as

let obj: any = "Hello World" as any

运算符

级联运算符

级联调用方法

obj = MyClass() 
  ..set_name("Hulo")
  ..set_age(18)
  ..print_info()

等价于:

obj = MyClass()
obj.set_name("Hulo")
obj.set_age(18)
obj.print_info()

级联修改属性

user = User()
  ..name = "Alice"
  ..age = 25
  ..email = "alice@example.com";

等价于：

user = User()
user.name = "Alice"
user.age = 25
user.email = "alice@example.com"

级联修改属性

list = []
  ..push(1)
  ..push(2)
  ..push(3);

等价于:

list = []
list.push(1)
list.push(2)
list.push(3)

空值合并

let name: str?
echo(name ?? "default") // default

name ??= "default"
echo(name) // default

name ??= "hulo"
echo(name) // default

条件成员访问

let name: str?
echo(name?.length) // name 为 null，不会调用length

三目运算符

let age = 18
echo(age >= 18 ? "Adult": "Minior") // Adult

扩展运算符

let list1 = [1, 2, 3]
let list2 = [0, ...list1, 4]
echo $list2 // [0, 1, 2, 3, 4]

箭头函数

fn printf(v: any) => echo(v)

管道运算符

ch <- "hello"

let data <- ch
